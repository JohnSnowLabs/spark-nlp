audioAssembler = AudioAssembler() \
	.setInputCol("audio_content") \
	.setOutputCol("audio_assembler")

     .setInputCols(["audio_assembler"]) \
     .setOutputCol("text")

pipelineModel = pipeline.fit(data)
pipelineDF = pipelineModel.transform(data)
