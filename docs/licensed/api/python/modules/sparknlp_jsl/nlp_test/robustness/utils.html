[docs]def calculate_proportion(f1_score: float) -> [float, None]:
    """
    This function calculates the proportion value for robustness based on F1 score.
    It assumes that noise_prob assigned during testing is set to the default value of 0.5.
    :param f1_score: f1_score of robustness testing results.
    :return: Corresponding proportion value for f1_score.
    """
    if f1_score > 0.9:
        return None
    elif f1_score > 0.75:
        return 0.05
    elif f1_score > 0.6:
        return 0.1
    elif f1_score > 0.4:
        return 0.2
    else:
        return 0.3




[docs]def suggest_perturbations(test_results: dict) -> dict:
    """
    This function calculates proportion value for each entity and each test types for robustness fixing.
    Calculation based on robustness test results and output format should be dictionary.
    :param test_results: Test result's metric dictionary obtained from robustness test.
    :return: Dictionary of suggestions containing test name, entity and proportion.
    """
    suggestions = {}
    for test, metrics in test_results['metrics'].items():
        proportions = {}
        for ent in metrics:
            if not ent in ['O', 'accuracy', 'macro avg', 'weighted avg', 'test', 'nan']:
                proportion_suggestion = calculate_proportion(metrics[ent]['f1-score'])
                if proportion_suggestion is not None:
                    proportions.update({ent: proportion_suggestion})
        if proportions != {}:
            suggestions.update({test: proportions})
    return suggestions
